    Anthropic format per RESEARCH.md Patterns 1-4:

    - **Endpoint:** `https://api.anthropic.com/v1/messages`
    - **Headers:**
      ```csharp
      request.Headers.Add("x-api-key", apiKey);
      request.Headers.Add("anthropic-version", "2023-06-01");
      // content-type set by StringContent constructor
      ```
      Do NOT use Authorization: Bearer for Claude. Use x-api-key header.

    - **Request body structure:**
      ```json
      {
        "model": "<ClaudeModel>",
        "max_tokens": 4096,
        "system": "<systemPrompt>",
        "messages": [...],
        "tools": [...]  // from GetToolsJson() (Anthropic format, per D-10)
      }
      ```
      Note: `system` is a top-level string field, NOT in the messages array.

    - **User message with screenshot (image content block):**
      ```csharp
      var userContent = new JArray();
      if (!string.IsNullOrEmpty(base64Png))
      {
          userContent.Add(new JObject
          {
              ["type"] = "image",
              ["source"] = new JObject
              {
                  ["type"] = "base64",
                  ["media_type"] = "image/png",
                  ["data"] = base64Png
              }
          });
      }
      userContent.Add(new JObject
      {
          ["type"] = "text",
          ["text"] = userMessage
      });
      var userMsg = new JObject { ["role"] = "user", ["content"] = userContent };
      ```

    - **Tool-use loop (max 10 iterations):**
      1. POST to endpoint
      2. On HTTP 429: return a special sentinel string `"__429__"` (do NOT set PendingResult here — the caller handles fallback)
      3. On HTTP 400/401/500 (per D-07): set PendingResult to `"[Error]: HTTP {code} — {body}"` and return
      4. Parse response JSON:
         - Check `stop_reason` field (not `done`)
         - If `stop_reason == "tool_use"`: iterate `content` array, find blocks where `type == "tool_use"`, extract `id`, `name`, `input`
         - Append the full assistant message (with its content array) to messages
         - Build tool result user message: `role: "user"`, content array of `tool_result` blocks:
           ```csharp
           var toolResults = new JArray();
           foreach (var block in contentArray)
           {
               if (block["type"]?.Value<string>() == "tool_use")
               {
                   string toolId = block["id"]!.Value<string>()!;
                   string toolName = block["name"]!.Value<string>()!;
                   string toolInput = block["input"]?.ToString(Formatting.None) ?? "{}";
                   Mod.Log.Info($"[ClaudeAPISystem] Tool call: {toolName}({toolInput})");
                   string result = m_ToolRegistry.Dispatch(toolName, toolInput);
                   Mod.Log.Info($"[ClaudeAPISystem] Tool result: {result}");
                   toolResults.Add(new JObject
                   {
                       ["type"] = "tool_result",
                       ["tool_use_id"] = toolId,
                       ["content"] = result
                   });
               }
           }
           messages.Add(new JObject { ["role"] = "user", ["content"] = toolResults });
           ```
         - Continue loop
         - If `stop_reason == "end_turn"`: extract text from content array (find block where `type == "text"`), set as PendingResult

    - **All writes to PendingResult use Interlocked.Exchange:**
      ```csharp
      Interlocked.Exchange(ref PendingResult, finalContent ?? "[Error]: Empty response.");
      ```

    - On any error: set PendingResult to `"[Error]: ..."` and return

    **6. Implement fallback logic in RunRequestAsync (D-06, D-07, D-08):**

    ```csharp
    private async Task RunRequestAsync(string userMessage, string? base64Png)
    {
        try
        {
            var setting = Mod.ActiveSetting;
            if (setting == null) { Interlocked.Exchange(ref PendingResult, "[Error]: Mod settings not loaded."); return; }

            string sysPrompt = setting.SystemPrompt ?? "";
            if (m_NarrativeMemory.IsInitialized)
                sysPrompt += m_NarrativeMemory.GetAlwaysInjectedContext();

            string apiKey = (setting.ClaudeApiKey ?? "").Trim();
            string model = (setting.ClaudeModel ?? "").Trim();

            // Log masked key
            string masked = apiKey.Length > 8 ? apiKey.Substring(0, 4) + "..." + apiKey.Substring(apiKey.Length - 4) : "(too short)";
            Mod.Log.Info($"[ClaudeAPISystem] Provider: Claude, model={model}, apiKey={masked}");

            string result = await RunClaudeRequestAsync(userMessage, base64Png, apiKey, model, sysPrompt).ConfigureAwait(false);

            if (result == "__429__")
            {
                // Rate limit fallback per D-06, D-08
                string ollamaBase = (setting.OllamaFallbackBaseUrl ?? "").Trim().TrimEnd('/');
                string ollamaModel = (setting.OllamaFallbackModel ?? "").Trim();

                if (string.IsNullOrEmpty(ollamaBase) || string.IsNullOrEmpty(ollamaModel))
                {
                    // D-08: no fallback configured
                    Interlocked.Exchange(ref PendingResult,
                        "\u26a0\ufe0f Rate limited by Claude. No Ollama fallback configured \u2014 set one up in mod settings.");
                    return;
                }

                // D-06: surface rate-limit notice to the chat panel via PendingResult
                // CityAgentUISystem's drain loop will consume this as an assistant message on the next frame
                Interlocked.Exchange(ref PendingResult, $"\u26a0\ufe0f Rate limited \u2014 retrying with {ollamaModel}...");
                Mod.Log.Info($"[ClaudeAPISystem] Claude 429 \u2014 falling back to Ollama {ollamaModel}");

                string ollamaKey = (setting.OllamaFallbackApiKey ?? "").Trim();
                await RunOllamaRequestAsync(userMessage, ollamaKey, ollamaModel, ollamaBase, sysPrompt).ConfigureAwait(false);
            }
            // else: result was set in RunClaudeRequestAsync (success or non-429 error)
        }
        catch (Exception ex)
        {
            Mod.Log.Error($"[ClaudeAPISystem] RunRequestAsync error: {ex}");
            Interlocked.Exchange(ref PendingResult, $"[Error]: {ex.Message}");
        }
        finally
        {
            m_RequestInFlight = false;
        }
    }
    ```
